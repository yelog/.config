配置详解

以下展示核心配置文件，您可以根据自己的需求进行调整

~/.config/karabiner/assets/complex_modifications/yelog.json 
{
  "title": "yelog key-maping perferences",
  "rules": [
    {
      // Caps Lock 按下时映射为 ⌘⌥⇧，单独按下切换大小写
      "description": "Change caps_lock to cmd+opt+shift if pressed with other keys",
      "manipulators": [
        {
          "type": "basic",
          "from": { "key_code": "caps_lock", "modifiers": { "optional": ["any"] } },
          "to": [
            { "key_code": "left_shift", "modifiers": ["left_command", "left_option"] }
          ],
          "to_if_alone": [
            { "hold_down_milliseconds": 100, "key_code": "caps_lock" }
          ]
        }
      ]
    },
    {
      // H/J/K/L 映射为方向键
      "description": "(cmd+opt+shift) + h/j/k/l == vim directional Keys",
      "manipulators": [
        { "from": { "key_code": "h", "modifiers": { "mandatory": ["left_shift", "left_command", "left_option"] } }, "to": [{ "key_code": "left_arrow" }], "type": "basic" },
        { "from": { "key_code": "j", "modifiers": { "mandatory": ["left_shift", "left_command", "left_option"] } }, "to": [{ "key_code": "down_arrow" }], "type": "basic" },
        { "from": { "key_code": "k", "modifiers": { "mandatory": ["left_shift", "left_command", "left_option"] } }, "to": [{ "key_code": "up_arrow" }], "type": "basic" },
        { "from": { "key_code": "l", "modifiers": { "mandatory": ["left_shift", "left_command", "left_option"] } }, "to": [{ "key_code": "right_arrow" }], "type": "basic" }
      ]
    }
  ]
}

💡 Karabiner-Elements 负责 Caps Lock 到 Hyper 键的底层映射。将此 JSON 文件放入 ~/.config/karabiner/assets/complex_modifications/ 目录，然后在 Karabiner-Elements 设置中启用即可。

~/.config/hammerspoon/ 
-- ============================================
-- init.lua - Hammerspoon 入口文件
-- ============================================

hs.hotkey.alertDuration = 0
hs.window.animationDuration = 0.125

-- 加载功能模块
require("modules.window.split")
require("modules.window.layout")
require("modules.window.position")
require("modules.window.resize")
require("modules.window.multiscreen")
require("modules.app")
require("modules.screen")


-- ============================================
-- config/keys.lua - 快捷键定义
-- ============================================

local M = {}

-- Hyper 键 (⌘⌥⇧) - 与 Karabiner 映射一致
M.hyper = { "shift", "alt", "cmd" }
-- Super 键 (⌘⌥⇧⌃)
M.super = { "shift", "alt", "cmd", "ctrl" }

return M


-- ============================================
-- config/apps.lua - 应用切换配置
-- ============================================

local keys = require("config.keys")

local M = {
  { prefix = keys.hyper, key = "V", message = "WeChat", appName = "WeChat" },
  { prefix = keys.hyper, key = "G", message = "QQ", appName = "QQ" },
  { prefix = keys.hyper, key = "F", message = "Finder", appName = "Finder" },
  { prefix = keys.hyper, key = "B", message = "Browser", appName = "Arc" },
  { prefix = keys.hyper, key = "I", message = "IntelliJ IDEA", appName = "IntelliJ IDEA" },
  { prefix = keys.hyper, key = "T", message = "Terminal", appName = "kitty" },
  { prefix = keys.hyper, key = ";", message = "ChatGPT", appName = "ChatGPT" },
  { prefix = keys.hyper, key = ",", message = "Codex", appName = "Codex" },
}

return M


-- ============================================
-- modules/app.lua - 应用切换模块
-- ============================================

local helper = require("utils.helper")
local apps = require("config.apps")

-- 自动解析 appName 获取 bundleId
local bundleIdCache = {}
local function resolveBundleId(appName)
  if bundleIdCache[appName] then return bundleIdCache[appName] end
  local output = hs.execute("osascript -e 'id of app \"" .. appName .. "\"' 2>/dev/null")
  if output and output ~= "" then
    local bundleId = output:gsub("%s+$", "")
    bundleIdCache[appName] = bundleId
    return bundleId
  end
  return nil
end

-- 绑定快捷键
hs.fnutils.each(apps, function(item)
  local bundleId = resolveBundleId(item.appName)
  hs.hotkey.bind(item.prefix, item.key, item.message, function()
    hs.application.launchOrFocusByBundleID(bundleId)
    helper.setMousePos()
  end)
end)

⚙️ 自定义配置

修改 config/keys.lua 定义快捷键前缀；修改 config/apps.lua 添加应用切换；修改 config/windows.lua 调整窗口管理。

应用使用 appName 而非 bundleId，更简洁易读。
